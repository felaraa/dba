SELECT ENTITY_CODE, ENTITY_NAME, X.PERIOD , X.DAY , X.AVG_0, X.STDDEV_0 FROM ( SELECT ENTITY_CODE, ENTITY_NAME, PERIOD, DAY, ROUND(AVG(M38673),4) AVG_0, ROUND(STDDEV_SAMP(M38673),4) STDDEV_0 FROM ( SELECT 'CN='||D_ENTITY_R4G_ENODEB.CN||';Municipio='||D_ENTITY_R4G_ENODEB.Municipio||';Operator='||D_ENTITY_R4G_ENODEB.Operator||';Region='||D_ENTITY_R4G_ENODEB.Region||';SiteType='||DA_R4G_SITE_ADD_INFO.SiteType||';Site_Name='||D_ENTITY_R4G_ENODEB.Site_Name||';Vendor='||D_ENTITY_R4G_ENODEB.Vendor||';eNodeBName='||D_ENTITY_R4G_ENODEB.eNodeBName ENTITY_CODE,D_ENTITY_R4G_ENODEB.CN||';'||D_ENTITY_R4G_ENODEB.Municipio||';'||D_ENTITY_R4G_ENODEB.Operator||';'||D_ENTITY_R4G_ENODEB.Region||';'||DA_R4G_SITE_ADD_INFO.SiteType||';'||D_ENTITY_R4G_ENODEB.Site_Name||';'||D_ENTITY_R4G_ENODEB.Vendor||';'||D_ENTITY_R4G_ENODEB.eNodeBName ENTITY_NAME,D_TIME.HOUR  PERIOD,D_TIME.DAY_OF_WEEK AS DAY , SUM(M38673) M38673 FROM F_R4G_ENODEB, D_TIME, D_ENTITY_R4G_ENODEB,DA_R4G_SITE_ADD_INFO WHERE D_TIME.DAY_OF_WEEK IN ('1','2','3','4','5','6','7') AND F_R4G_ENODEB.AA_r4g_site_add_info_key=DA_R4G_SITE_ADD_INFO.AA_r4g_site_add_info_key AND F_R4G_ENODEB.entity_key=D_ENTITY_R4G_ENODEB.entity_key AND D_ENTITY_R4G_ENODEB.entity_type_name = 'R4G_eNodeB' AND D_ENTITY_R4G_ENODEB.entity_key = F_R4G_ENODEB.entity_key  GROUP BY DA_R4G_SITE_ADD_INFO.SiteType, D_ENTITY_R4G_ENODEB.Region, D_ENTITY_R4G_ENODEB.Operator, D_ENTITY_R4G_ENODEB.Vendor, D_ENTITY_R4G_ENODEB.CN, D_ENTITY_R4G_ENODEB.Municipio, D_ENTITY_R4G_ENODEB.eNodeBName, D_ENTITY_R4G_ENODEB.Site_Name,D_TIME.REPORT_DATE_TIME,D_TIME.HOUR ,D_TIME.DAY_OF_WEEK) GROUP BY ENTITY_CODE, ENTITY_NAME, PERIOD, DAY) X WHERE ( X.AVG_0 IS NOT NULL ) ORDER BY ENTITY_CODE, X.DAY, X.PERIOD
